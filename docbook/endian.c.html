#ifndef ENDIAN_SOURCE
#define ENDIAN_SOURCE

#include "../tools/memory.h"

void endian (void * memory, size_t extent) 

{
	register byte * first = (byte *)(memory);
	register byte * final = first + extent;
	while (first < final) 
	{
		register byte byte = *first;
		*first++ = *--final;
		*final = byte;
	}
	return;
}

#endif
